This card uses separate header, content and footer components for more complex layouts.

 Primary action Secondary action
View Code 
<!-- Basic Card with title attribute and href -->
<gcds-card card-title="Card with Title Attribute" href="https://example.com">
  <gcds-card-content>
    <p>This is a basic card with a title attribute and href. The card component is used to group related content and actions.</p>
    <gcds-button>Learn more</gcds-button>
  </gcds-card-content>
</gcds-card>

<!-- Card with header, content and footer components and href -->
<gcds-card href="https://example.com" card-title="Card with Header Component">
  <gcds-card-header>Card with Header Component</gcds-card-header>
  <gcds-card-content>
    <p>This card uses separate header, content and footer components for more complex layouts.</p>
  </gcds-card-content>
  <gcds-card-footer>
    <gcds-button button-role="primary">Primary action</gcds-button>
    <gcds-button button-role="secondary">Secondary action</gcds-button>
  </gcds-card-footer>
</gcds-card>
